# Stage 1: Get the tools and absctl from the official tools image
FROM aerospike/aerospike-tools:latest AS tools-collector

# Stage 2: Build the final image
FROM aerospike/aerospike-server:latest

# Install basic dependencies for Python-based tools (asadm/asinfo)
RUN apt-get update && apt-get install -y \
    python3 \
    python3-setuptools \
    libssl-dev \
    curl \
    && rm -rf /var/lib/apt/lists/*

# Copy the tools (aql, asadm, asinfo, asbackup, etc.) from the tools image
COPY --from=tools-collector /usr/bin/aql /usr/bin/aql
COPY --from=tools-collector /usr/bin/asadm /usr/bin/asadm
COPY --from=tools-collector /usr/bin/asinfo /usr/bin/asinfo
COPY --from=tools-collector /usr/bin/asbackup /usr/bin/asbackup
COPY --from=tools-collector /usr/bin/asrestore /usr/bin/asrestore
COPY --from=tools-collector /usr/lib/python3/dist-packages/aerospike_* /usr/lib/python3/dist-packages/

# Install absctl directly via the official GitHub release (more reliable than the artifact server)
RUN ARCH=$(uname -m) && \
    if [ "$ARCH" = "aarch64" ]; then ABS_ARCH="arm64"; else ABS_ARCH="amd64"; fi && \
    curl -L -o /usr/local/bin/absctl "https://github.com/aerospike/aerospike-backup-service/releases/latest/download/absctl-linux-${ABS_ARCH}" && \
    chmod +x /usr/local/bin/absctl

EXPOSE 3000 3001 3002

CMD ["asd", "--foreground"]
